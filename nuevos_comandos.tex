\makeatletter

\renewenvironment{proof}[1][\proofname]{\par
	\pushQED{\qed}%
	\normalfont \topsep6\p@\@plus6\p@\relax
	\trivlist
	\item[\hskip\labelsep
	            \itshape
	            #1\@addpunct{.}]\mbox{}\\*
}{
	\popQED\endtrivlist\@endpefalse
}

\newenvironment{enefecto}[1][En efecto,]{\par
	\pushQED{\qed}%
	\normalfont \topsep6\p@\@plus6\p@\relax
	\trivlist
	\item[\hskip\labelsep
	            #1]\ignorespaces % <--- \ignorespaces mantiene el texto seguido
}{%
	\popQED\endtrivlist\@endpefalse
}

\makeatother
